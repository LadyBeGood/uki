function isPrime(num) {
    if (num <= 1) {
        return false;
    }
    if (num === 2) {
        return true;
    }
    if (num % 2 === 0) {
        return false;
    }

    for (let i = 3; i <= Math.sqrt(num); i += 2) {
        if (num % i === 0) {
            return false;
        }
    }

    return true;
}

function printPrimes(num1, num2) {
    console.log(`Prime numbers from ${num1} to ${num2} are:`);

    for (let num = num1; num <= num2; num++) {
        if (isPrime(num)) {
            console.log(num);
        }
    }
}
